```rust
#[tokio::test]
async fn test_five_level_nested_child_contexts() {
    let (mut ctx, calls, _ops) = MockDurableContext::new().build().await;

    let result: i32 = ctx
        .child_context("level1", |mut l1| async move {
            let r1: Result<i32, String> = l1.step("l1_step", || async { Ok(1) }).await?;
            assert_eq!(r1.unwrap(), 1);

            let inner: i32 = l1
                .child_context("level2", |mut l2| async move {
                    let r2: Result<i32, String> = l2.step("l2_step", || async { Ok(2) }).await?;
                    assert_eq!(r2.unwrap(), 2);

                    let inner: i32 = l2
                        .child_context("level3", |mut l3| async move {
                            let r3: Result<i32, String> = l3.step("l3_step", || async { Ok(3) }).await?;
                            assert_eq!(r3.unwrap(), 3);

                            let inner: i32 = l3
                                .child_context("level4", |mut l4| async move {
                                    let r4: Result<i32, String> = l4.step("l4_step", || async { Ok(4) }).await?;
                                    assert_eq!(r4.unwrap(), 4);

                                    let inner: i32 = l4
                                        .child_context("level5", |mut l5| async move {
                                            let r5: Result<i32, String> = l5.step("l5_step", || async { Ok(5) }).await?;
                                            Ok(r5.unwrap())
                                        })
                                        .await?;
                                    Ok(inner + r4.unwrap()) // Hmm, r4 already consumed
                                    // Actually, store values before consuming:
                                })
                                .await?;
                            Ok(inner)
                        })
                        .await?;
                    Ok(inner)
                })
                .await?;
            Ok(inner)
        })
        .await
        .unwrap();
```

Use a cleaner pattern: each level's step returns a value, and the innermost level (level 5) returns a sum or identifiable value. The key assertion is that all 5 levels complete without panics and produce correct results. Simplify: each child_context returns its level number, and the final result is the deepest level's value (5).

**Assertions:**

**Important:** The closure for `child_context` receives an owned `DurableContext` and must return `Result<T, DurableError>`. The inner step calls return `Result<Result<T, E>, DurableError>` — unwrap the outer `?` then `.unwrap()` the inner.

**test_parallel_in_child_in_parallel** — Use `MockDurableContext::new().build().await` (execute mode). Create a 3-level nesting structure:
- Level 1: `parallel` with 2 branches
- Level 2: Each branch creates a `child_context`
- Level 3: Inside each child_context, run another `parallel` with 2 branches

```rust
#[tokio::test]
async fn test_parallel_in_child_in_parallel() {
    let (mut ctx, _calls, _ops) = MockDurableContext::new().build().await;

    // Level 1: outer parallel with 2 branches
    type OuterBranch = Box<
        dyn FnOnce(DurableContext) -> Pin<Box<dyn Future<Output = Result<i32, DurableError>> + Send>>
            + Send,
    >;

    let branches: Vec<OuterBranch> = vec![
        Box::new(|mut branch_ctx| {
            Box::pin(async move {
                // Level 2: child context inside parallel branch
                let child_result: i32 = branch_ctx
                    .child_context("child_a", |mut child_ctx| async move {
                        // Level 3: inner parallel inside child context
                        type InnerBranch = Box<
                            dyn FnOnce(DurableContext) -> Pin<Box<dyn Future<Output = Result<i32, DurableError>> + Send>>
                                + Send,
                        >;
                        let inner_branches: Vec<InnerBranch> = vec![
                            Box::new(|mut inner_ctx| {
                                Box::pin(async move {
                                    let r: Result<i32, String> = inner_ctx.step("inner_a1", || async { Ok(10) }).await?;
                                    Ok(r.unwrap())
                                })
                            }),
                            Box::new(|mut inner_ctx| {
                                Box::pin(async move {
                                    let r: Result<i32, String> = inner_ctx.step("inner_a2", || async { Ok(20) }).await?;
                                    Ok(r.unwrap())
                                })
                            }),
                        ];
                        let inner_result = child_ctx.parallel("inner_parallel_a", inner_branches, ParallelOptions::new()).await?;
                        // Sum the inner results
                        let sum: i32 = inner_result.results.iter()
                            .filter_map(|item| item.result.as_ref())
                            .sum();
                        Ok(sum)
                    })
                    .await?;
                Ok(child_result)
            })
        }),
        Box::new(|mut branch_ctx| {
            Box::pin(async move {
                // Level 2: child context in second branch
                let child_result: i32 = branch_ctx
                    .child_context("child_b", |mut child_ctx| async move {
                        // Level 3: inner parallel in second child
                        type InnerBranch = Box<
                            dyn FnOnce(DurableContext) -> Pin<Box<dyn Future<Output = Result<i32, DurableError>> + Send>>
                                + Send,
                        >;
                        let inner_branches: Vec<InnerBranch> = vec![
                            Box::new(|mut inner_ctx| {
                                Box::pin(async move {
                                    let r: Result<i32, String> = inner_ctx.step("inner_b1", || async { Ok(100) }).await?;
                                    Ok(r.unwrap())
                                })
                            }),
                            Box::new(|mut inner_ctx| {
                                Box::pin(async move {
                                    let r: Result<i32, String> = inner_ctx.step("inner_b2", || async { Ok(200) }).await?;
                                    Ok(r.unwrap())
                                })
                            }),
                        ];
                        let inner_result = child_ctx.parallel("inner_parallel_b", inner_branches, ParallelOptions::new()).await?;
                        let sum: i32 = inner_result.results.iter()
                            .filter_map(|item| item.result.as_ref())
                            .sum();
                        Ok(sum)
                    })
                    .await?;
                Ok(child_result)
            })
        }),
    ];

    let result = ctx
        .parallel("outer_parallel", branches, ParallelOptions::new())
        .await
        .unwrap();

    // Outer parallel has 2 branches — both should succeed
    assert_eq!(result.results.len(), 2);
    for item in &result.results {
        assert_eq!(item.status, BatchItemStatus::Succeeded);
    }

    // Collect results (order may vary due to tokio::spawn non-determinism)
    let mut values: Vec<i32> = result.results.iter()
        .filter_map(|item| item.result.as_ref().copied())
        .collect();
    values.sort();
    // Branch A: 10 + 20 = 30, Branch B: 100 + 200 = 300
    assert_eq!(values, vec![30, 300]);
}
```

**Key considerations:**
- The type alias for branch closures must use `Box<dyn FnOnce(DurableContext) -> Pin<Box<dyn Future<...> + Send>> + Send>` because parallel requires `Send + 'static`.
- Inner type aliases can be declared inside the closure body or at function scope — declare at function scope if the compiler requires it, otherwise inline is fine.
- The parallel branch ordering is non-deterministic (tokio::spawn). Sort results before comparing values. Use `values.sort()` then compare with sorted expected values.
- Each child_context returns the sum of its inner parallel branch results. The outer parallel collects these sums.
- The test completing successfully proves that 3-level nesting works with correct ID namespacing (no collisions between branches or nesting levels).
