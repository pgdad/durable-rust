    1. Add import at top of file: `use std::time::Duration;`

    2. Replace the current `handle.await.map_err(...)` block with timeout-aware version:
       ```rust
       let name_owned = name.to_string();
       let mut handle = tokio::spawn(async move { f().await });
       let user_result = if let Some(secs) = options.get_timeout_seconds() {
           match tokio::time::timeout(Duration::from_secs(secs), &mut handle).await {
               Ok(join_result) => join_result.map_err(|join_err| {
                   DurableError::checkpoint_failed(
                       &name_owned,
                       std::io::Error::other(format!("step closure panicked: {join_err}")),
                   )
               })?,
               Err(_elapsed) => {
                   handle.abort();
                   return Err(DurableError::step_timeout(&name_owned));
               }
           }
       } else {
           handle.await.map_err(|join_err| {
               DurableError::checkpoint_failed(
                   &name_owned,
                   std::io::Error::other(format!("step closure panicked: {join_err}")),
               )
           })?
       };
       ```
       NOTE: `handle` must be `mut` for `&mut handle` reference. `tokio::time::timeout` on `&mut JoinHandle` works because `&mut JoinHandle` implements Future.

    3. In the `Err(error)` arm of the checkpoint match, add predicate check BEFORE the `current_attempt <= max_retries` check (per FEAT-14: "fail immediately without consuming retry budget"):
       ```rust
       Err(error) => {
           let max_retries = options.get_retries().unwrap_or(0);

           // Check retry predicate first — false means fail immediately
           let should_retry = if let Some(pred) = options.get_retry_if() {
               pred(error as &dyn std::any::Any)
           } else {
               true // no predicate — retry all errors (backward compatible)
           };

           if should_retry && (current_attempt as u32) <= max_retries {
               // RETRY checkpoint (existing code unchanged)
           }
           // FAIL checkpoint (existing code unchanged)
       }
       ```
       IMPORTANT: The `error` binding from `Err(error)` is `&E`. Passing `error as &dyn std::any::Any` requires `E: 'static`, which is already satisfied by the method's `E: DeserializeOwned + Send + 'static` bound.

If step.rs doesn't have a test module, add tests to tests/e2e/tests/ instead (Task 2 in Plan 02 will handle comprehensive e2e tests).
